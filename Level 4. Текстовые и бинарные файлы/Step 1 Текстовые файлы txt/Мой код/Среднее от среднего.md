В файле _data.txt_, доступном вашему решению, находится несколько строк с числами, записанными через пробел. Найдите среднее арифметическое чисел, больших среднего арифметического всех чисел файла и выведите его с точностью не менее 4 знаков после запятой.

## Пример

| Ввод                                                                 | Вывод   |
| -------------------------------------------------------------------- | ------- |
| -100 59 83<br>-6 80 26 2 -96<br>-20 -81 74 -34<br>-94 66 -3 3 77 -11 | 52.2222 |

f = open('data.txt')
s = map(str.strip, f.readlines())
d = []
for line in s:
    d.extend(line.split())
d = list(map(int, d))
d_avarage = round(sum(d) / len(d), 4)
d_filter = list(filter(lambda x: x > d_avarage, d))  # числа больше среднего
d_d = round(sum(d_filter) / len(d_filter), 4)
print(d_d)
