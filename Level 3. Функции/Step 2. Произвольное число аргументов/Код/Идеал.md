Из множества слов всегда можно найти идеальное: в нём больше, чем в других, различных букв; если количество таких букв одинаково, то оно длиннее других; а если и по длине одинаково, то больше по алфавиту.

Напишите функцию _ideal()_, принимающую произвольное число слов и возвращающую выбранный из них идеал. При подсчёте количества различных букв регистр не учитывайте.

## Пример

|Ввод|Вывод|
|---|---|
|print(ideal('Spring', 'morning', 'Time'))|morning|
def ideal(*n):
    maxim = 0
    dic = {}
    lst = []
    for i in n:  # сортировка по количеству уникальных букв
        t = set(i.lower())
        maxim = max(len(t), maxim)
        dic[i] = dic.get(i, len(t))
    for k, v in dic.items():
        if v == maxim:
            lst.append(k)
    if len(lst) > 1:
        dic = {}
        maxim = 0
        for i in lst:
            maxim = max(len(i), maxim)
            dic[i] = dic.get(i, len(i))
        lst = []
        for k, v in dic.items():
            if v == maxim:
                lst.append(k)
    if len(lst) > 1:
        dic = {}
        maxim = ''
        for i in lst:
            maxim = max(i, maxim)
            dic[i] = dic.get(i, i)
        lst = []
        for k, v in dic.items():
            if v == maxim:
                lst.append(k)
    n = lst[0]
    return n
