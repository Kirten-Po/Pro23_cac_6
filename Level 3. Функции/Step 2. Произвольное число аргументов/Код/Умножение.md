Напишите функцию _multiplier()_, принимающую произвольное число аргументов-чисел и возвращающую произведение всех ненулевых значений. Если таких не нашлось или аргументы не переданы, верните _None_.

## Пример

|Ввод|Вывод|
|---|---|
|print(multiplier(0, 1, 2, 0, 5))|10|

def multiplier(*args):
    import math
    lst = [arg for arg in args if arg != 0]
    if lst:
        return math.prod(lst)
    return
