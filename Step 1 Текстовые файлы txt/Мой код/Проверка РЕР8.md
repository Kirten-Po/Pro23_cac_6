def check():
    with open('solve.py', encoding='utf-8') as solve:
        opera = ['+', '=', '-', '/', '*']
        bracets = ['(', ')']
        end = [' ', ':', ')', ']', '\n']
        need_space = False
        prev_space = 0
        first = False
        second = False
        tri = False
        for line in solve:
            idx = len(line) - 1
            cur_space = len(line) - len(line.lstrip())
            if cur_space % 4 != 0:
                tri = True
            if need_space:  
                if cur_space <= prev_space:
                    tri = True
                need_space = False
            if line.rstrip().endswith(':'):
                need_space = True
                prev_space = cur_space
            for i, item in enumerate(line):
                if item in opera:
                    if i == 0 or i == idx:
                        first = True
                    if line[i - 1] != ' ' or line[i + 1] != ' ':
                        first = True
                elif item in bracets:
                    if i < idx:
                        if line[i - 1] == ' ' or line[i + 1] == ' ':
                            second = True
                        elif item == ')':
                            if line[i + 1] not in end:
                                second = True
                            
    if first:
        return -1
    elif second:
        return -2
    elif tri:
        return -3
    else:
        return 0
