from math import floor


f = open('rose.txt')
s = [x.rstrip() for x in f.readlines()]
p_all = len(s)  # всего пикслей
d = []
for line in s:  # перевод строк в списки и строки в числа
    a = list(map(int, line.split()))
    d.append(a)
p_enough = list(filter(lambda x: x[0] > 220, d))
p_result = floor((len(p_enough) / p_all) * 100)
print(f'{p_result}%')
