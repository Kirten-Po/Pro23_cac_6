from sys import stdin


line = stdin.readlines()
line1 = [x.strip() for x in line]
lst_result = []
for item in line1:
    with open(item, encoding='utf-8') as dil:
        lst_1 = [x.strip() for x in dil]
        lst_result.extend(lst_1)
with (open('strings.txt', 'w', encoding='utf-8') as strings):
    result = sorted(lst_result, key=lambda x: (-len(x), x))
    print(*result, sep='\n', file=strings)
