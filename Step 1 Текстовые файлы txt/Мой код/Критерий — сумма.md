input_file = open('input.txt')
result_file = open('result.txt', 'w')
all_numbers = []
for line in input_file:
    all_numbers.extend(map(int, line.split()))
number_sum = round(sum(all_numbers), 0)
if number_sum % 2 == 0:
    finish_number = filter(lambda x: x % 2 == 0, all_numbers)
else:
    finish_number = filter(lambda x: x % 2 != 0, all_numbers)
result_num = sorted(finish_number)
print(*result_num, sep='\n', file=result_file)
