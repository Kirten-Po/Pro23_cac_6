f = open('statistics.txt', 'w', encoding='utf-8')
d = open('data.txt')
d1 = sorted([int(x) for x in d.readlines()])
medium = len(d1) // 2
print(f'Наибольшее число: {max(d1)}', file=f)
print(f'Наименьшее число: {min(d1)}', file=f)
print(f'Количество уникальных чисел: {len(set(d1))}', file=f)
print(f'Среднее арифметическое: {round(sum(d1) / len(d1), 2)}', file=f)
if len(d1) % 2 != 0:
    print(f'Медиана: {round(d1[medium], 2)}', file=f)
else:
    print(f'Медиана: {round((d1[medium - 1] + d1[medium]) / 2, 2)}', file=f)
