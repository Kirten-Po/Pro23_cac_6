import csv


with (open('sunspots.csv', encoding='utf-8') as file,
      open('result.txt', 'w', encoding='utf-8') as result):
    reader = csv.DictReader(file, delimiter='.')
    dic = {}
    for item in reader:
        dic[item['date']] = dic.get(item['date'], []) + [int(item['number'])]
    print(min(dic, key=dic.get), file=result)
    print(max(dic, key=dic.get), file=result)
