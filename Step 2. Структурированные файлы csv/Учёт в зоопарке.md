import csv
from pprint import pprint


with open('zoo.csv') as file:
    reader = csv.DictReader(file, delimiter=',')
    dic = {}
    for item in reader:
        mode = item['mode']
        name = item['name']
        age = item['age']
        dic[item['aviary']] = dic.get(item['aviary'], []) + [f'{mode} {name}, {age}']
        dic[item['aviary']].sort()
    pprint(dic, sort_dicts=False)
