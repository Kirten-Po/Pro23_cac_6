import csv


with (open('road.csv', encoding='utf-8') as file,
      open('report.txt', 'w', encoding='utf-8') as result):
    reader = csv.DictReader(file, delimiter=',')
    lst = []
    for item in reader:
        time = int(item['time2']) - int(item['time1'])
        speed = (5 / time) * 3600
        if speed > 90:
            lst.append(item['no'])
    print(*lst, sep='\n', file=result)
